CREATE TABLE FIXED(ACCNO NUMBER (5), YEARS NUMBER (4), AMOUNT NUMBER (6), INTEREST NUMBER (3));
INSERT INTO FIXED VALUES (:ACCNO, :YEARS, :AMOUNT, :INTEREST);
DECLARE
CURSOR CUR IS SELECT * FROM fixed;
BEGIN
FOR I IN CUR
LOOP
IF I.YEARS<=2 THEN 
UPDATE fixed SET INTREST=I.AMOUNT*0.09*(1/12) WHERE ACCNO=I.ACCNO;
ELSIF
I.YEARS>=2 AND I.YEARS<=5 THEN 
UPDATE fixed SET INTREST=I.AMOUNT*0.11*(1/12) WHERE ACCNO=I.ACCNO;
ELSE
UPDATE fixed SET INTREST=I.AMOUNT*0.15*(1/12) WHERE ACCNO=I.ACCNO;
END IF;
END LOOP;
END
