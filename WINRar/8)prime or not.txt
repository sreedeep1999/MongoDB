DECLARE
N NUMBER (20); 
I NUMBER (20);
M NUMBER(20); 

BEGIN

N : = : N; 
FOR I IN 1..N

LOOP

IF (N MOD I=0) THEN
M := M+1;
END IF;

END LOOP; 

IF (N>0)
THEN
DBMS_OUTPUT.PUT_LINE(N | | ' NUMBER IS PRIME');

ELSE
DBMS_OUTPUT.PUT_LINE(N | |' NUMBER IS NOT PRIME');
END IF;

END
