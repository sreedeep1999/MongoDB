begin
update accse tbalance=balance-amount1wherec_no=ac_no1;
end;
create or replace procedure depositt (ac_no1 innumber,amount1 innumber)is
begin
update accset balance=balance+amount1whereac_no=ac_no1;
end;

choice number; ac_no1 number(5); amount number(5);

else

end;
